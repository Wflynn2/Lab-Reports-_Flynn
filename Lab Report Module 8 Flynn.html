import numpy as np import matplotlib.pyplot as plt from numba import jit # JIT-optimized function to generate the Julia Set @jit(nopython=True) def julia_set(c, xmin, xmax, ymin, ymax, width, height, max_iter): r1 = np.linspace(xmin, xmax, width) r2 = np.linspace(ymin, ymax, height) n3 = np.empty((width, height)) for i in range(width): for j in range(height): z = r1[i] + 1j * r2[j] n = 0 while abs(z) <= 2 and n < max_iter: z = z*z + c n += 1 n3[i, j] = n return n3 # Parameters for the plot xmin, xmax, ymin, ymax = -1.5, 1.5, -1.5, 1.5 width, height = 800, 800 # Image resolution max_iter = 256 # Number of iterations for accuracy # Choose a value for the complex constant c c = complex(-0.8, 0.156) # Generate the Julia Set n3 = julia_set(c, xmin, xmax, ymin, ymax, width, height, max_iter) # Plotting the Julia Set plt.figure(figsize=(10, 10)) plt.imshow(n3.T, extent=[xmin, xmax, ymin, ymax], cmap='hot', interpolation='bilinear') plt.colorbar() plt.title(f"Julia Set for c = {c}") plt.show()
