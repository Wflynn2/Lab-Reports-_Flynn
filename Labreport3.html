import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d.art3d import Poly3DCollection
import numpy as np

# Function to calculate the midpoints of a triangle
def midpoint(p1, p2, p3):
    return (p1 + p2 + p3) / 3

# Recursive function to generate the Sierpinski triangle in 3D
def sierpinski_3d(vertices, depth):
    if depth == 0:
        # Create a single triangle at depth 0
        triangles = [vertices]
        return triangles
    else:
        # Calculate midpoints of the current triangle
        p1, p2, p3 = vertices
        m1 = midpoint(p1, p2, p3)
        m2 = midpoint(p1, p2, p3)
        m3 = midpoint(p1, p2, p3)
        
        # Generate smaller triangles at the next depth level
        smaller_triangles = []
        smaller_triangles += sierpinski_3d([p1, m1, m2], depth - 1)
        smaller_triangles += sierpinski_3d([p2, m2, m3], depth - 1)
        smaller_triangles += sierpinski_3d([p3, m3, m1], depth - 1)
        
        return smaller_triangles

# Initial triangle vertices in 3D
p1 = np.array([1, 0, 0])
p2 = np.array([-1, 0, 0])
p3 = np.array([0, 1, 0])
depth = 3

# Create the Sierpinski triangle fractal
triangles = sierpinski_3d([p1, p2, p3], depth)

# Plotting the result
fig = plt.figure()
ax = fig.add_subplot(111, projection='3d')

# Plot each triangle
for tri in triangles:
    ax.add_collection3d(Poly3DCollection([tri], color='cyan', linewidths=1, edgecolors='r', alpha=0.3))

ax.set_xlabel('X')
ax.set_ylabel('Y')
ax.set_zlabel('Z')

plt.show()

Conclusion

Explanation of the code:

  • midpoint(): This function calculates the midpoint of a triangle by averaging the coordinates of its three vertices.
  • sierpinski_3d(): This recursive function generates the 3D Sierpiński triangle. At each recursion depth, it subdivides the triangle into smaller triangles, removing the central portion to create the fractal structure.
  • p1, p2, p3: These represent the coordinates of the three corners of the initial equilateral triangle from which the fractal starts.
  • depth: This parameter controls the recursion depth, determining how many iterations of subdivision will occur. A higher depth results in a more detailed fractal structure.
  • Poly3DCollection: This is a feature from the Matplotlib library used to visualize the 3D Sierpiński triangles by rendering the smaller triangles as 3D polygons.
