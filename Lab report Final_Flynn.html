Introduction - Part 1: Chaos Game

The Chaos Game, according to Michael Barnsley of the Georgia Institute of Technology, is very simple to play but has an extraordinary outcome. He began with a large sheet of paper with three points marked on it. The coordinates were (1,2), (3,4), (5,6). Then he chose a point at random on the paper, which is his starting point. Then he rolled a dice and if he came up with a four, he would go halfway between his random point and the coordinate with four. He would continue this method for a long period, and the result was extraordinary; it produced the Sierpinski triangle. The Barnsley Fern is self-similar, just like the Sierpinski triangle. It can also be created with the Chaos Game, by choosing a random point and using four special functions that move the points to the right, left, rotates and flip the points.

import matplotlib.pyplot as plt
from random import random, randint
import numpy as np

def interpolate(P, Q, ratio=0.5):
    """Move from P toward Q by a given ratio."""
    return (ratio*P[0] + (1-ratio)*Q[0], ratio*P[1] + (1-ratio)*Q[1])

# Set up vertices: try regular polygons (e.g., triangle, square, pentagon, etc.)
def generate_regular_polygon(sides=3, radius=2, center=(2, np.sqrt(3))):
    angles = np.linspace(0, 2*np.pi, sides, endpoint=False)
    return [(center[0] + radius*np.cos(a), center[1] + radius*np.sin(a)) for a in angles]

vertices = generate_regular_polygon(sides=5)  # Try with 5 sides (pentagon)

iterates = 50000
x, y, colors = [0]*iterates, [0]*iterates, [0]*iterates

# Random starting point
x[0], y[0] = random()*4, random()*4

# Ratio (not just 0.5)
ratio = 0.5

for i in range(1, iterates):
    k = randint(0, len(vertices) - 1)
    x[i], y[i] = interpolate(vertices[k], (x[i-1], y[i-1]), ratio=ratio)
    colors[i] = k  # Save vertex chosen for color coding

# Plotting
plt.figure(figsize=(8, 8))
plt.scatter(x, y, c=colors, cmap='hsv', s=0.1)  # Colorful points based on vertex
plt.title('Generalized Chaos Game with Polygon (5 sides)')
plt.axis('equal')
plt.axis('off')  # Hide axis for aesthetics
plt.show()

# Function to draw a 3D-like fractal tree using recursion
def fractal_tree_3d(length, depth, angle):
    if depth == 0:
        return
    else:
        forward(length)
        left(angle)
        fractal_tree_3d(length * 0.7, depth - 1, angle)  # Left branch
        right(2 * angle)
        fractal_tree_3d(length * 0.7, depth - 1, angle) # Right branch

        left(angle)
        backward(length)

        # Simulate depth by slightly shifting the branches
        if depth > 2: # Avoid shifting too early
          penup()
          right(angle/2)
          forward(depth * 2)
          pendown()
          left(angle/2)
          fractal_tree_3d(length * 0.5, depth - 2, angle) # depth branch
          penup()
          backward(depth * 2)
          left(angle/2)
          pendown()
          right(angle/2)
