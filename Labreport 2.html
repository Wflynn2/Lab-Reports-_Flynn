# Define the blip_line function (Koch curve)
def blip_line(length, order):
    if order == 0:
        forward(length)
    else:
        blip_line(length / 3, order - 1)
        left(60)
        blip_line(length / 3, order - 1)
        right(120)
        blip_line(length / 3, order - 1)
        left(60)
        blip_line(length / 3, order - 1)

# Draw the Koch snowflake
def koch_snowflake(length, order):
    for _ in range(3):
        blip_line(length, order)
        right(120)

# Set up the turtle
jump(250, 500)
